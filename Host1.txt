# Set details
$SERVERNAME = "Host1"


# Set the IP address for the Host

Rename-Computer -NewName $SERVERNAME
Get-NetIPAddress
New-NetIPAddress -InterfaceIndex 16 -IPAddress 192.168.0.10 -DefaultGateway "192.168.0.1" -PrefixLength 24 
Restart-Computer


#Install Hyper-V Feature
Install-WindowsFeature -Name Hyper-V -ComputerName Host1 -IncludeManagementTools -Restart
